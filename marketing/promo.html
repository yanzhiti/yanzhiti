🚀 快速开始

# 克隆仓库
git clone https://github.com/yanzhiti/yanzhiti.git
cd yanzhiti

# 创建虚拟环境
python -m venv venv
source venv/bin/activate # Windows: venv\Scripts\activate

# 安装依赖
pip install -e ".[dev]"

# 运行衍智体
export YANZHITI_API_KEY=your-api-key
yanzhiti
